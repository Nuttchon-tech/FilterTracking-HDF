Apps Script Code (v3 — Real-time Sync)
// ============================================================ // Filter Tracker — Google Apps Script v3 (Real-time Sync) // Deploy เป็น Web App: Execute as Me | Who has access: Anyone // ============================================================ var LOG_SHEET = "Filter_Log"; var STATE_SHEET = "Filter_State"; // ── GET: ดึง shared state กลับมาที่แอป ────────────────────── function doGet(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(STATE_SHEET); if (!sheet || sheet.getLastRow() < 2) { return buildResponse({ status: "ok", state: null, message: "No state yet" }); } var json = sheet.getRange(2, 1).getValue(); var state = JSON.parse(json); return buildResponse({ status: "ok", state: state }); } catch(err) { return buildResponse({ status: "error", message: err.toString() }); } } // ── POST: รับ action log + อัปเดต shared state ────────────── function doPost(e) { try { var payload = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var logSheet = getOrCreateLogSheet(ss); if (payload.action !== "SAVE_STATE") { logSheet.appendRow([ payload.timestamp || new Date().toLocaleString("th-TH"), payload.action || "—", payload.machine || "—", payload.tankFrom || payload.tankId || "—", payload.tankTo || "—", payload.tankType || "—", payload.operator || "—", payload.filterType || "—", payload.qty || 0, payload.stockBefore !== undefined ? payload.stockBefore : "—", payload.stockAfter !== undefined ? payload.stockAfter : "—", payload.receiveDate || "—", payload.note || "—" ]); } if (payload.sharedState) { var stateSheet = getOrCreateStateSheet(ss); stateSheet.getRange(2, 1).setValue( JSON.stringify(payload.sharedState) ); stateSheet.getRange(2, 2).setValue( new Date().toLocaleString("th-TH") ); } return buildResponse({ status: "ok", rows: logSheet.getLastRow() }); } catch (err) { return buildResponse({ status: "error", message: err.toString() }); } } function getOrCreateLogSheet(ss) { var sheet = ss.getSheetByName(LOG_SHEET); if (!sheet) { sheet = ss.insertSheet(LOG_SHEET); var h = sheet.getRange(1,1,1,13); h.setValues([["Timestamp","Action","Machine","Tank (From/ID)", "Tank To","Tank Type","Operator","Filter Type", "Qty Used","Stock Before","Stock After","Receive Date","Note"]]); h.setBackground("#1a1d27"); h.setFontColor("#3b82f6"); h.setFontWeight("bold"); sheet.setFrozenRows(1); } return sheet; } function getOrCreateStateSheet(ss) { var sheet = ss.getSheetByName(STATE_SHEET); if (!sheet) { sheet = ss.insertSheet(STATE_SHEET); sheet.getRange(1,1).setValue("state_json"); sheet.getRange(1,2).setValue("last_updated"); sheet.getRange(1,1,1,2).setFontWeight("bold") .setBackground("#1a1d27").setFontColor("#10b981"); } return sheet; } function buildResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
