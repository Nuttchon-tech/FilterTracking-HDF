Apps Script Code (วางใน Google Apps Script)
วิธีใช้:
1. เปิด Google Sheet → Extensions → Apps Script
2. ลบ Code เดิม แล้ววาง Code ด้านล่างทั้งหมด
3. กด Deploy → New deployment → Web App
  • Execute as: Me  |  Who has access: Anyone
4. Copy URL → ใส่ในช่อง Script URL ด้านบน
// ============================================ // Filter Tracker — Google Apps Script // วางทั้งหมดนี้ใน Apps Script แล้ว Deploy // ============================================ function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:"ok", msg:"Filter Tracker API ready"})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var sheetId = data.sheetId; // ถ้าระบุ Sheet ID ให้ใช้ Sheet นั้น if (sheetId && sheetId.length > 10) { ss = SpreadsheetApp.openById(sheetId); } // สร้าง Sheet ถ้ายังไม่มี var logSheet = ss.getSheetByName("Filter_Log"); if (!logSheet) { logSheet = ss.insertSheet("Filter_Log"); logSheet.appendRow([ "Timestamp","Machine","Tank ID","Tank Type", "Action","Operator","Filter Type","Qty Used", "Stock Before","Stock After","Note","Receive Date" ]); // จัดรูปแบบหัวตาราง var header = logSheet.getRange(1,1,1,12); header.setBackground("#1a1d27"); header.setFontColor("#3b82f6"); header.setFontWeight("bold"); logSheet.setFrozenRows(1); } // เพิ่มแถวข้อมูล logSheet.appendRow([ data.timestamp || new Date().toLocaleString("th-TH"), data.machine || "—", data.tankId || "—", data.tankType || "—", data.action || "—", data.operator || "—", data.filterType || "—", data.qty || 0, data.stockBefore !== undefined ? data.stockBefore : "—", data.stockAfter !== undefined ? data.stockAfter : "—", data.note || "—", data.receiveDate || "—" ]); return ContentService .createTextOutput(JSON.stringify({status:"ok", rows: logSheet.getLastRow()})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error", msg: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
